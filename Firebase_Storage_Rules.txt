// Recommended Firebase Storage Rules for BookReader

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow users to read/write their own books
    match /books/{userId}/{allPaths=**} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && 
                      request.auth.uid == userId &&
                      request.resource.size < 100 * 1024 * 1024 && // 100MB limit
                      request.resource.contentType.matches('application/pdf');
    }
    
    // Alternative: More permissive rules for testing
    // match /{allPaths=**} {
    //   allow read, write: if request.auth != null;
    // }
  }
}

// To apply these rules:
// 1. Go to Firebase Console
// 2. Navigate to Storage
// 3. Click on "Rules" tab
// 4. Replace existing rules with the above
// 5. Click "Publish"
